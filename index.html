import tkinter as tk from tkhtmlview import HTMLLabel root = tk.Tk() root.title("HTML в Tkinter") root.geometry("800x600") # HTML-код, который будет отображён html_content = """

Привет из HTML!

Это текст с жирным и курсивом.

""" # Создаём виджет HTMLLabel и передаём HTML html_label = HTMLLabel( root, html=html_content, # HTML-код для рендеринга width=100, # Ширина в символах height=30 # Высота в строках ) html_label.pack(padx=20, pady=20, fill="both", expand=True) # Размещаем в окне root.mainloop()
